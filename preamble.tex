\NewDocumentCommand{\LstTikzBox}{mO{.949\linewidth}mO{}}%
{%
\newsavebox{#1}%
\begin{lrbox}{#1}%
  \begin{tikzpicture}[every node/.style={inner sep=0,outer sep=0}]%
    \node{%
      \hspace*{18pt}%
      \begin{minipage}{#2}%
        \vspace*{-2pt}
        \lstinputlisting[%
          style=myEnvStyle,%
          language=Rust,%
          nolol=ture,%
          #4%
        ]{#3}%
        \vspace*{-6pt}
      \end{minipage}%
      \hspace*{4pt}%
    };%
  \end{tikzpicture}%
\end{lrbox}%
}
\NewDocumentCommand{\LstTikzFig}{ommmO{}}%
{%
  \IfNoValueTF{#1}{%
    \edef\tmp{\csname#4\endcsname}%
    \expandafter\LstTikzBox\tmp{#2}[#5]%
    \begin{figure}[ht]%
      \captionsetup{type=lstlisting}%
      \centering%
      \expandafter\usebox{\tmp}%
      \caption{#3}\label{#4}%
    \end{figure}%
  }{%
    \lstinputlisting[%
      style=myEnvStyle,%
      language=Rust,%
      nolol=ture,%
      caption={#3},
      label={#4},
      #5%
    ]{#2}%
  }%
}
